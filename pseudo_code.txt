void main()
{
 LwAr ar = new LwAr();
 Renderer renderer = new OpenGLRenderer();
 ar.renderer = renderer;
 
 // ID 0, resolution 640 x 480, 30 fps
 Camera cam = new Camera(0, 640, 480, 30);
 ar.cam = cam;
 
 // Callbacks
 ar.OnUpdate += OnUpdate(); 
 ar.OnDraw += OnDraw();
 
 3dObject cube = new 3dObject(3DObject.Cube);
 Marker marker = new Marker();
 
 // starts everything
 ar.start();
}

void OnUpdate()
{
 Transform transform = ar.FindMarker(marker);
 cube.SetTransform(transform);
}

void OnDraw()
{
 ar.DrawObject(cube);
 
 // manuelles zeichenn vom Background, damit es noch verändert werden kann
 ar.DrawBackground(cam.frame);
}
